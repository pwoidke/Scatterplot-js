Scatterplot.js


Default implementation

            $(document).ready(function () {
                $('#default').scatter();
                $('#point').plot();
            });

Typical implementation

            $(document).ready(function () {
                $('#demo').scatter({ height: 500, width: 500, xLabel: 'X-Label Test', yLabel: 'Y-Label Test', rows: 5, columns: 5, subsections: 5 });
                $('#demoPoint1').plot({ xPos: 100, yPos: 100, color: 'red' });
                $('#demoPoint2').plot({ xPos: 320, yPos: 145, radius: 20, color: 'blue' });
                $('#demoPoint3').plot({ xPos: 460, yPos: 410, radius: 10, color: 'green' });
            });
